<script> const circle = document.querySelector(".progress-ring__circle"); const radius = circle.r.baseVal.value; const circumference = 2 * Math.PI * radius; const score_in_points = document.querySelector(".score_in_points"); score_in_points.addEventListener("change", function () { setProgress(score_in_points); }); circle.style.strokeDasharray = `${circumference} ${circumference}`; circle.style.strokeDashoffset = circumference; function setProgress(percent) { const offset = circumference - (percent / 100) * circumference; circle.style.strokeDashoffset = offset; } setProgress(score_in_points.textContent); </script>
